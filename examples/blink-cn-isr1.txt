decimal
14 constant LED
variable blinkcnt
variable cnt10ms

: cn-isr 
  ?pinchange if \ *** press
    dup
    1 = if
      blinkcnt @ 1 > if
        -1 blinkcnt +!
      then
    then
    17 = if
      1 blinkcnt +!
    then
  else         \ *** release
    drop
  then
;


: blink-isr
  1 cnt10ms +!
  cnt10ms @ blinkcnt @ >= if
    0 cnt10ms !
    LED d@ 1 xor LED d!
  then
;


: cn-isr-init
  1 pintocn
  17 pintocn
  ISR_PINCHANGE ['] cn-isr setisr
;

: blink-isr-init
  1 LED pm
  0 cnt10ms !
  10 blinkcnt !
  ISR_CORETIMER ['] blink-isr setisr
;


: test
  cn-isr-init
  blink-isr-init
  ei
;
